26point3 random_in_unit_disk() {
27 static std::uniform_real_distribution<double> distribution(0.0, 1.0);
28 static std::mt19937 generator; // Mersenne Twister
29
30 double theta = 2.0 * M_PI * distribution(generator);
31 double r = sqrt(distribution(generator));
32
33 double x = r * cos(theta);
34 double y = r * sin(theta);
35
36 return point3(x, y, 0.0); // Assumindo que o disco está no plano XY

100 void initialize() {
101 image_height = static_cast<int>(image_width / aspect_ratio);
102 image_height = (image_height < 1) ? 1 : image_height;
103
104 center = lookfrom;
105
106 // Determine viewport dimensions.
107 auto theta = degrees_to_radians(vfov);
108 auto h = tan(theta/2);
109 auto viewport_height = 2 * h * focus_dist;
110 auto viewport_width = viewport_height * (static_cast<double>(image_width)/image_height);
111
112 // Calculate the u,v,w unit basis vectors for the camera coordinate frame.
113 w = unit_vector(lookfrom - lookat);
114 u = unit_vector(cross(vup, w));
115 v = cross(w, u);
116
117 // Calculate the vectors across the horizontal and down the vertical viewport edges.
118 vec3 viewport_u = viewport_width * u; // Vector across viewport horizontal edge
119 vec3 viewport_v = viewport_height * -v; // Vector down viewport vertical edge
120
121 // Calculate the horizontal and vertical delta vectors to the next pixel.
122 pixel_delta_u = viewport_u / image_width;
123 pixel_delta_v = viewport_v / image_height;
124
125 // Calculate the location of the upper left pixel.
126 auto viewport_upper_left = center - (focus_dist * w) - viewport_u/2 - viewport_v/2;
127 pixel00_loc = viewport_upper_left + 0.5 * (pixel_delta_u + pixel_delta_v);
128
129 // Calculate the camera defocus disk basis vectors.
130 auto defocus_radius = focus_dist * tan(degrees_to_radians(defocus_angle / 2));
131 defocus_disk_u = u * defocus_radius;
132 defocus_disk_v = v * defocus_radius;
133 }

142 ray get_ray(int i, int j) const {
143 auto pixel_center = pixel00_loc + (i * pixel_delta_u) + (j * pixel_delta_v);
144 auto pixel_sample = pixel_center + pixel_sample_square();
145
146 auto ray_origin = (defocus_angle <= 0) ? center : defocus_disk_sample();
147 auto ray_direction = pixel_sample - ray_origin;
148
149 return ray(ray_origin, ray_direction);
150 }

157 vec3 pixel_sample_square() const {
158 auto px = -0.5 + random_double();
159 auto py = -0.5 + random_double();
160 return (px * pixel_delta_u) + (py * pixel_delta_v);
161 }

168 point3 defocus_disk_sample() const {
169 auto p = random_in_unit_disk();
170 return center + (p[0] * defocus_disk_u) + (p[1] * defocus_disk_v);

181 color ray_color(const ray& r, int depth, const hittable& world) const {
182 if (depth <= 0)
183 return color(0,0,0);
184
185 hit_record rec;
186
187 if (world.hit(r, interval(0.001, infinity), rec)) {
188 ray scattered;
189 color attenuation;
190 if (rec.mat->scatter(r, rec, attenuation, scattered))
191 return attenuation * ray_color(scattered, depth-1, world);
192
193 return color(0,0,0);
194 }
195
196 vec3 unit_direction = unit_vector(r.direction());
197 auto a = 0.5*(unit_direction.y() + 1.0);
198 return (1.0-a)*color(1.0, 1.0, 1.0) + a*color(0.5, 0.7, 1.0);
199 }
