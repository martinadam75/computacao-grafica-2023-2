157vec3 refract(const vec3& uv, const vec3& normal, double etai_over_etat) {
158 auto cos_theta = fmin(dot(-uv, normal), 1.0);
159 vec3 r_out_perp = etai_over_etat * (uv + cos_theta * normal);
160 vec3 r_out_parallel = -sqrt(fabs(1.0 - r_out_perp.length_squared())) * normal;
161 return r_out_perp + r_out_parallel;

185 bool scatter(const ray& r_in, const hit_record& rec, color& attenuation, ray& scattered)
186 const override {
187 attenuation = color(1.0, 1.0, 1.0);
188 double refraction_ratio = rec.front_face ? (1.0/ir) : ir;
189
190 vec3 unit_direction = unit_vector(r_in.direction());
191 double cos_theta = fmin(dot(-unit_direction, rec.normal), 1.0);
192 double sin_theta = sqrt(1.0 - cos_theta*cos_theta);
193
194 bool cannot_refract = refraction_ratio * sin_theta > 1.0;
195 vec3 direction;
196
197 if (cannot_refract || reflectance(cos_theta, refraction_ratio) > random_double())
198 direction = reflect(unit_direction, rec.normal);
199 else
200 direction = refract(unit_direction, rec.normal, refraction_ratio);
201
202 scattered = ray(rec.p, direction);
203 return true;
204 }
205
206 private:
207 double ir;
216 static double reflectance(double cosine, double ref_idx) {
217 auto r0 = (1-ref_idx) / (1+ref_idx);
218 r0 = r0*r0;
219 return r0 + (1-r0)*pow((1 - cosine),5);
220 }
